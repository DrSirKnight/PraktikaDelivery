 package com.example.delivery.service;
 
 import com.example.delivery.model.City;
 import com.example.delivery.DeliveryException;
 import org.springframework.stereotype.Service;
 
 import java.util.ArrayList;
 import java.util.HashMap;
 import java.util.List;
 
 @Service
 public class CityService {
     List<City> cityList;
 
     CityService() {
         cityList = new ArrayList<>();
         HashMap<String, Double> vehicles = new HashMap<>();
         vehicles.put("car", 4.0);
         vehicles.put("scooter", 3.5);
         vehicles.put("bike", 3.0);
         cityList.add(new City("tallinn", "Tallinn-Harku", vehicles));
         vehicles = new HashMap<>();
         vehicles.put("car", 3.5);
         vehicles.put("scooter", 3.0);
         vehicles.put("bike", 2.5);
         cityList.add(new City("tartu", "Tartu-Tõravere", vehicles));
         vehicles = new HashMap<>();
         vehicles.put("car", 3.0);
         vehicles.put("scooter", 2.5);
         vehicles.put("bike", 2.0);
         cityList.add(new City("pärnu", "Pärnu", vehicles));
     }
 
     public City getCity(String cityName) throws DeliveryException {
         for (City city : cityList) {
             if (city.getName().equals(cityName)) {
                 return city;
             }
         }
         throw new DeliveryException("Invalid city name");
     }
 }
