 package com.example.delivery.service;
 
 import com.example.delivery.model.Weather;
 import com.example.delivery.repository.WeatherRepository;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.scheduling.annotation.Scheduled;
 import org.springframework.stereotype.Service;
 import org.xml.sax.SAXException;
 
 import javax.xml.parsers.ParserConfigurationException;
 import java.io.IOException;
 import java.util.List;
 
 @Service
 public class WeatherService {
     private final WeatherRepository repository;
     private final WebService web;
 
     private final String[] stations = {"Tallinn-Harku", "Tartu-Tõravere", "Pärnu"};
 
     @Autowired
     public WeatherService(WeatherRepository repository, WebService web) throws IOException, ParserConfigurationException, SAXException {
         this.repository = repository;
         this.web = web;
         updateWeatherData();
     }
 
     public Weather getWeatherByStation(String station) {
         return repository.findByLatestStation(station);
     }
 
     @Scheduled(cron = "0 15 * * * ?")
     public void updateWeatherData() throws IOException, ParserConfigurationException, SAXException {
         List<Weather> weatherList = web.getWebData(stations);
         repository.saveAll(weatherList);
     }
 
     public String[] getStations() {
         return stations;
     }
 }
