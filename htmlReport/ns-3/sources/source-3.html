 package com.example.delivery.service;
 
 import com.example.delivery.model.Weather;
 import org.springframework.stereotype.Service;
 import org.w3c.dom.Document;
 import org.w3c.dom.Element;
 import org.w3c.dom.Node;
 import org.w3c.dom.NodeList;
 import org.xml.sax.SAXException;
 
 import javax.xml.parsers.DocumentBuilder;
 import javax.xml.parsers.DocumentBuilderFactory;
 import javax.xml.parsers.ParserConfigurationException;
 import java.io.IOException;
 import java.net.URL;
 import java.net.URLConnection;
 import java.time.LocalDateTime;
 import java.util.ArrayList;
 import java.util.Arrays;
 import java.util.List;
 
 @Service
 public class WebService {
     public List<Weather> getWebData(String[] stations) throws IOException, ParserConfigurationException, SAXException {
         URL url = new URL("https://www.ilmateenistus.ee/ilma_andmed/xml/observations.php");
 
         URLConnection conn = url.openConnection();
         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
         DocumentBuilder builder = factory.newDocumentBuilder();
         Document doc = builder.parse(conn.getInputStream());
 
         NodeList observations = doc.getElementsByTagName("station");
 
         ArrayList<Weather> weatherList = new ArrayList<>();
 
         LocalDateTime localDate = LocalDateTime.now();
         for (int i = 0; i < observations.getLength(); i++) {
             Node observation = observations.item(i);
             if (observation.getNodeType() == Node.ELEMENT_NODE) {
                 Element observationElement = (Element) observation;
                 String stationName = observationElement.getElementsByTagName("name").item(0).getTextContent();
                 if (Arrays.asList(stations).contains(stationName)) {
                     String phenomenon = observationElement.getElementsByTagName("phenomenon").item(0).getTextContent();
                     String wmoCodeStr = observationElement.getElementsByTagName("wmocode").item(0).getTextContent();
                     String airTemperatureStr = observationElement.getElementsByTagName("airtemperature").item(0).getTextContent();
                     String windSpeedStr = observationElement.getElementsByTagName("windspeed").item(0).getTextContent();
                     int wmoCode = 0;
                     if (!wmoCodeStr.isEmpty()) {
                         wmoCode = Integer.parseInt(wmoCodeStr);
                     }
                     double airTemperature = strToDouble(airTemperatureStr);
                     double windSpeed = strToDouble(windSpeedStr);
                     weatherList.add(new Weather(stationName, wmoCode, airTemperature, windSpeed, phenomenon, localDate));
                 }
             }
         }
         return weatherList;
     }
 
     private double strToDouble(String str) {
         double num;
         try {
             num = Double.parseDouble(str);
         } catch (NumberFormatException nfe) {
             num = 0.0;
         }
         return num;
     }
 }
